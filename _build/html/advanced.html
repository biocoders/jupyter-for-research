Widgets and Dashboards#

from ipywidgets import Button, IntText, Label, VBox, HBox

button = Button()
button.description = "Click me!"

output = IntText()
output.value = 0

def update_counter(evt):
    output.value += 1

button.on_click(update_counter)
HBox([Label("Number of clicks"), output, button])
